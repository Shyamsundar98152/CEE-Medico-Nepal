Google Apps Script Backend Setup

Why is this required?

Web browsers enforce strict security protocols (CORS) that prevent uploading directly to Google Drive. This custom script acts as a safe middleware proxy allowing direct uploads.

Step-by-Step Implementation

1

Open Google Apps Script Dashboard

Visit Google Apps Script Dashboard and click the New Project button on the top left.

2

Copy & Paste the Generated Code

Replace any pre-populated template placeholder code inside the script editor screen with this custom tailored API controller: 

const TARGET_FOLDER_ID = '17OKeKZ4nRCOlrj6_7u_wVrEJDu2o8Rzn'; // Hardcoded destination target

function doGet(e) {
  return HtmlService.createHtmlOutput('
Connected successfully.
')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const result = handleDirectUpload(data.data, data.filename, data.subject, data.chapter, data.manualFolder);
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function handleDirectUpload(data, filename, subject, chapter, manualFolder) {
  try {
    let parentFolder = DriveApp.getFolderById(TARGET_FOLDER_ID);
    
    // Dynamic hierarchal directory setup
    if (subject && subject.trim() !== '') {
      parentFolder = getOrCreateFolder(parentFolder, subject.trim());
    }
    if (chapter && chapter.trim() !== '') {
      parentFolder = getOrCreateFolder(parentFolder, chapter.trim());
    }
    if (manualFolder && manualFolder.trim() !== '') {
      parentFolder = getOrCreateFolder(parentFolder, manualFolder.trim());
    }
    
    // Convert base64 stream to functional binary file blob
    const mimeMatch = data.match(/^data:(.*);base64,/);
    const contentType = mimeMatch ? mimeMatch[1] : 'application/octet-stream';
    const cleanData = data.substring(data.indexOf(',') + 1);
    
    const blob = Utilities.newBlob(Utilities.base64Decode(cleanData), contentType, filename);
    const file = parentFolder.createFile(blob);
    
    return { success: true, url: file.getUrl(), name: file.getName() };
  } catch (error) {
    return { success: false, error: error.toString() };
  }
}

function getOrCreateFolder(parent, name) {
  const folders = parent.getFoldersByName(name);
  if (folders.hasNext()) {
    return folders.next();
  }
  return parent.createFolder(name);
}
3

Deploy the Web App

On the top right of your Apps Script Editor, click Deploy > New deployment. Select Web app (via the gear icon) and fill out:

  • Execute as: Me (your-email@gmail.com)
  • Who has access: Anyone
4

Authorize Permissions & Bind URL

Click Deploy. Approve the Google Permissions warning window. Copy the generated Web app URL from the final screen and paste it into the API Connection Settings card on the left panel.
